// --- Keyboard ---
bool IsKeyPressed(int key);                  // Pressed once
bool IsKeyPressedRepeat(int key);            // Pressed again (repeat)
bool IsKeyDown(int key);                     // Being pressed
bool IsKeyReleased(int key);                 // Released once
bool IsKeyUp(int key);                       // NOT being pressed
int GetKeyPressed(void);                     // Get keycode, call multiple times for queue
int GetCharPressed(void);                    // Get unicode char, call multiple times for queue
void SetExitKey(int key);                    // Default is ESC

void SetGamepadVibration(int gamepad, float leftMotor, float rightMotor, float duration);

bool IsMouseButtonUp(int button);
int GetMouseX(void);

void SetMouseCursor(int cursor);
